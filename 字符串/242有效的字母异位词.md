class Solution {
    public boolean isAnagram(String s, String t) {
        char[] chars = new char[26];
        for (char c : s.toCharArray()) {
            chars[c - 'a']++;
        }
        for (char c : t.toCharArray()) {
            chars[c - 'a']--;
        }

        for (char aChar : chars) {
            if (aChar != 0){
                return false;
            }
        }
        return true;

    }
}

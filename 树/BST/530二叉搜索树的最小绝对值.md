class Solution {
    public int minAbs = Integer.MAX_VALUE;
    public TreeNode preNode;
    public int getMinimumDifference(TreeNode root) {
        inOrder(root);
        return minAbs;
    }
    public void inOrder(TreeNode root){
        if (root == null) return;
        inOrder(root.left);
        if (preNode != null){
            minAbs = Math.min(minAbs,root.val - preNode.val);
        }
        preNode = root;
        inOrder(root.right);
    }
}

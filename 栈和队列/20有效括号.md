public boolean isValid(String s) {

    Stack<Character> stack = new Stack<>();
    for (char c : s.toCharArray()) {
        if (c == '(' || c == '[' || c=='{'){
            stack.push(c);
        }else{
            if (stack.isEmpty()){
                return false;
            }
            char pop = stack.pop();
            boolean result1 = c == ')' && pop != '(';
            boolean result2 = c == ']' && pop != '[';
            boolean result3 = c == '}' && pop != '{';

            if (result1 || result2 || result3){
                return false;
            }
        }
    }
    return stack.isEmpty();
}

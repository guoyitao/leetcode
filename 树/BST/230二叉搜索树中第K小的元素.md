class Solution {
    int count;
    int val;
    public int kthSmallest(TreeNode root, int k) {
        run(root,k);
        return val;
    }

    public void run(TreeNode root,int k){
        if (root == null) return;
        run(root.left,k);
        count++;
        if (count == k){
            val = root.val;
            return;
        }
        run(root.right,k);
    }
}
